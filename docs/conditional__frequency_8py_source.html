23 def calc_cond_freq(model_name, path, params):
24  """
25  This function counts frequency of given params out of the csv file specified by path, then calculate an estimated
26  conditional probability using these counts. Currently only support naive params(use , as and) with bool value.
27  TODO: implement composite params handler.
28  :param model_name: name of model, used when load/save as model_name.pickle
29  :param path: absolute path to the input csv file
30  :param params: target1|given1 target2|given2 target3|given3 target4|given4 ...
31  :return: a dict of conditional probabilities { "target1|given1":0.001, "target2|given2":0.001,
32  "target3|given3":0.001, "target4|given4":0.001, ...}
33  """
34  result = {}
35  param_list = params.split(" ")
36  d = pd.read_csv(path)
37  cols = list(d.columns.values)
38  for p in param_list:
39  data = d.copy()
40  targets = p.split("|")[0].split(",")
41  given = p.split("|")[1].split(",")
42  for g in given:
43  if g not in cols:
44  logging("COND_FREQ_HDL", "[ ERROR ] %s (from param %s) not exists in data cols" % (g, p))
45  continue
46  data = data[data[g] == True]
47  given_cnt = len(data.index)
48  for t in targets:
49  if t not in cols:
50  logging("COND_FREQ_HDL", "[ ERROR ] %s (from param %s) not exists in data cols" % (t, p))
51  continue
52  data = data[data[t] == True]
53  target_cnt = len(data.index)
54  result[p] = target_cnt / given_cnt if given_cnt != 0 else 0
55  logging("COND_FREQ_HDL", "[ INFO ] %s applied_%s" % (model_name, path.split("/")[-1].split(".")[0]))
56 
57  return result
